47 inline float32 RandomFloat(float32 lo, float32 hi)
48 {
49  float32 r = (float32)(rand() & (RAND_LIMIT));
50  r /= RAND_LIMIT;
51  r = (hi - lo) * r + lo;
52  return r;
53 }
