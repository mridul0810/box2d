161  // This class sets the group flags to b2_particleGroupCanBeEmpty so that

186  // dt is seconds that have passed, particleIndices is an optional pointer
187  // to an array which tracks which particles have been created and
188  // particleIndicesCount is the size of the particleIndices array.
189  // This function returns the number of particles created during this
190  // simulation step.
191  int32 Step(const float32 dt, int32* const particleIndices,
192  const int32 particleIndicesCount)
193  {
194  b2Assert(m_particleSystem);
195  int32 numberOfParticlesCreated = 0;
196  // How many (fractional) particles should we have emitted this frame?
197  m_emitRemainder += m_emitRate * dt;
198 
199  b2ParticleDef pd;
200  pd.color = m_color;
201  pd.flags = m_flags;
202  pd.group = m_group;
203 
204  // Keep emitting particles on this frame until we only have a
205  // fractional particle left.
206  while (m_emitRemainder > 1.0f) {
207  m_emitRemainder -= 1.0f;
208 
209  // Randomly pick a position within the emitter's radius.
210  const float32 angle = Random() * 2.0f * b2_pi;
211  // Distance from the center of the circle.
212  const float32 distance = Random();
213  b2Vec2 positionOnUnitCircle(sin(angle), cos(angle));
214 
215  // Initial position.
216  pd.position.Set(
217  m_origin.x + positionOnUnitCircle.x * distance * m_halfSize.x,
218  m_origin.y + positionOnUnitCircle.y * distance * m_halfSize.y);
219  // Send it flying
220  pd.velocity = m_startingVelocity;
221  if (m_speed != 0.0f)
222  {
223  pd.velocity += positionOnUnitCircle * m_speed;
224  }
225 
226  const int32 particleIndex = m_particleSystem->CreateParticle(pd);
227  if (m_callback)
228  {
229  m_callback->ParticleCreated(m_particleSystem, particleIndex);
230  }
231  if (particleIndices &&
232  numberOfParticlesCreated < particleIndicesCount)
233  {
234  particleIndices[numberOfParticlesCreated] = particleIndex;
235  }
236  ++numberOfParticlesCreated;
237  }
238  return numberOfParticlesCreated;
239  }

242  // Calculate a random number 0.0f..1.0f.
243  static float32 Random()
244  {
245  return ((float32)rand() / (float32)RAND_MAX);
246  }
