46  // Value of a particle parameter.

58  // Particle parameter definition.
59  struct Definition
60  {
61  // Values associated with this particle parameter definition.
62  const Value *values;
63  // Number of entries in the "values" array.
64  uint32 numValues;
65 
66  // Calculate the mask (bitwise OR) of all values in the definition.
67  uint32 CalculateValueMask() const
68  {
69  uint32 mask = 0;
70  for (uint32 i = 0; i < numValues; i++)
71  {
72  mask |= values[i].value;
73  }
74  return mask;
75  }
76  };
