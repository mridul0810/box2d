41 // Determine whether the specified position is within the bounding
42 // box that contains the arrow returning the identifier of this arrow
43 // or notSelectedIdentifier if the position is outside of the bounding
44 // box.
45 uint32 Arrow::Hit(const b2Vec2 &position, uint32 notSelectedIdentifier) const
46 {
47  // Calculate the extent of the arrow's bounding box in viewport
48  // coordinates.
49  const float32 arrowExtent =
50  CalculateScale() * Arrow::k_size * 0.5f;
51  b2Vec2 arrowPosition;
52  // Translate position into this arrow's space.
53  const b2Vec2 relativePosition =
54  position - *CalculateViewportPosition(&arrowPosition);
55  // Determine whether position is within the arrow's bounding box.
56  if (fabs(relativePosition.x) < arrowExtent &&
57  fabs(relativePosition.y) < arrowExtent)
58  {
59  return m_identifier;
60  }
61  return notSelectedIdentifier;
62 }

85 // Calculate the scaling factor for the arrow given the current
86 // viewport extents.
87 float32 Arrow::CalculateScale() const
88 {
89  // Determine the minimum of a viewport dimension. Extents are
90  // defined as half the viewport width / height so multiply by
91  // 2.0f to get the complete dimension.
92  const float32 minDimension =
93  std::min(m_extents->x, m_extents->y) * 2.0f;
94  // Calculate the scaling factor for the arrow in world space.
95  return (minDimension / Arrow::k_size) * m_scale;
96 }

98 // Calculate the viewport position of the arrow.
99 b2Vec2* Arrow::CalculateViewportPosition(
100  b2Vec2 * const viewportPosition) const
101 {
102  b2Assert(viewportPosition);
103  viewportPosition->x =
104  m_viewCenter->x + (m_extents->x * m_position.x);
105  viewportPosition->y =
106  m_viewCenter->y + (m_extents->y * m_position.y);
107  return viewportPosition;
108 }

110 // Draw a colored arrow rotated by angle, scaled by scale and at
111 // the viewport relative position using DebugDraw.
112 // With no transformation matrix applied, the arrow is drawn in box
113 // area (3.5f, 3.5f) (see Arrow::k_size) and the overall bounding box
114 // of the arrow is (-1.75f, -1.75f) to (1.75f, 1.75f).
115 void Arrow::DrawArrow(const b2Color &color, const float32 angle,
116  const float32 scale, const b2Vec2 &position)
117 {
118  static const b2Vec2 square[] =
119  {
120  b2Vec2(0.25f, 1.0f), b2Vec2(0.25f, -1.0f),
121  b2Vec2(-1.75f, -1.0f), b2Vec2(-1.75f, 1.0f)
122  };
123  static const b2Vec2 triangle[] =
124  {
125  b2Vec2(0.25f, 1.75f), b2Vec2(1.75f, 0.0f),
126  b2Vec2(0.25f, -1.75f)
127  };
128  // Build the transformation matrix.
129  glPushMatrix();
130  glTranslatef(position.x, position.y, 0.0f);
131  glRotatef(angle, 0.0f, 0.0f, 1.0f);
132  glScalef(scale, scale, 1.0f);
133  // Draw the arrow.
134  DebugDraw dbgDraw;
135  dbgDraw.DrawFlatPolygon(square, B2_ARRAY_SIZE(square), color);
136  dbgDraw.DrawFlatPolygon(triangle, B2_ARRAY_SIZE(triangle), color);
137  glPopMatrix();
138 }
