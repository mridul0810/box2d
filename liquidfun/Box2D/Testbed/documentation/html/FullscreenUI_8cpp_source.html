69 // Process mouse events and perform arrow selection returning the
70 // ID of the selected arrow if it has been clicked.
71 uint32 FullscreenUI::Mouse(const int32 button, const int32 state,
72  const int32 previousState,
73  const b2Vec2 &mousePosition)
74 {
75  if (GetEnabled() && button == GLUT_LEFT_BUTTON)
76  {
77  // If the mouse button is being pressed, update the current
78  // arrow selection.
79  if (state == GLUT_DOWN && previousState != GLUT_DOWN)
80  {
81  for (uint32 i = 0; i < k_numArrows &&
82  m_selection == e_SelectionNone; ++i)
83  {
84  m_selection = s_arrows[i].Hit(mousePosition,
85  e_SelectionNone);
86  }
87  }
88  // If the mouse button has been released, return the current
89  // arrow selection.
90  if (state == GLUT_UP)
91  {
92  const uint32 selectedArrow = m_selection;
93  m_selection = e_SelectionNone;
94  return selectedArrow;
95  }
96  }
97  return e_SelectionNone;
98 }

111 // Set the view parameters for all widgets in the UI.
112 void FullscreenUI::SetViewParameters(const b2Vec2 *viewCenter,
113  const b2Vec2 *extents)
114 {

116  {

123 // Draw s_arrows.
124 void FullscreenUI::DrawArrows()
125 {
126  for (uint32 i = 0; i < k_numArrows; ++i)
127  {
128  const Arrow &arrow = s_arrows[i];
129  const uint32 id = arrow.GetIdentifier();
130  if ((id != e_SelectionParameterPrevious &&
131  id != e_SelectionParameterNext) ||
132  m_particleParameterSelectionEnabled)
133  {
134  arrow.Draw(m_selection);
135  }
136  }
137 }

139 // Draw text at the bottom of the screen.
140 void FullscreenUI::DrawFooterText(const std::string &text)
141 {
142  glColor4f(1, 1, 1, 1);
143  glPushMatrix();
144  glLoadIdentity();
145  if (m_aspectRatio > 1)
146  {
147  glScalef(1 / m_aspectRatio, 1, 1);
148  }
149  else
150  {
151  glScalef(1, m_aspectRatio, 1);
152  }
153  glScalef(0.04f, 0.04f, 1);
154  glTranslatef(0, -20, 0);
155  glTranslatef(text.length() * -0.75f, -4, 0);
156  glScalef(0.015f, 0.015f, 1);
157  glutStrokeString(GLUT_STROKE_MONO_ROMAN,
158  (const unsigned char *)text.c_str());
159  glPopMatrix();
160 }
